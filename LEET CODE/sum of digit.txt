
int sum (int num)
{   int su=0;
    while (num)
    {
        su+=num%10;
        num/=10;
    }
    return su;
}
class Solution {
public:
    int addDigits(int num) {
      while (num/10)
      {
        num=sum(num);
      }
      return num;
    }
};
