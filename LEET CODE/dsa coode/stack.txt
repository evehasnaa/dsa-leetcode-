#include <iostream>
using namespace std;
struct stack {
    
    int arr[4];
    int top=-1;
    bool isempty (){
      return top==-1;
        
    }
    void puch(int value)
    {
       // if(isempty())
         if(top<3)
        {
            top++;
            arr[top]=value;
        }
        else cout<<"stack underflaw\n";
        //}
        
    }
    void pop (){
        if(!isempty()){
            top--;
        }
        else cout<<"stack ovflaw\n";
    }
    int peak(){
        if(isempty()) return -1;
        else
        return arr[top];
    }

};
int main()
{
stack s;
s.puch(55);
s.puch(44);

s.puch(10);
   while (!s.isempty()){
       cout<<s.peak()<<endl;

   }
