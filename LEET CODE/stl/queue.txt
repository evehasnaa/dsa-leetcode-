    #include <iostream>
    #include<queue>
    using namespace std;

    int main()
    {
        queue<int>q;
        int t;
        cin>>t;
        while(t--)
        {
            int op,n=0;
            cin >>op;
        if (op==1)
            {   cin>>n;
                q.push(n);
            }
        else if ( op==2 && q.size()>0 )
                q.pop();
        else if (op==3 && q.size()>0)
             cout <<q.front()<<endl;
             else if(op==3)
                cout <<"Empty!"<<endl;
        }
        return 0;
    }
