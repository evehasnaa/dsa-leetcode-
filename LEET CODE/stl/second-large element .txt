#include <iostream>
#include<vector>
#include<set>
using namespace std;

void init(){
    cin.tie(0);
    cin.sync_with_stdio(0);
}
int main()
{   init();
    int t;
    cin>>t;
    while(t--)
    {

        set<int>s;
        int n;
        cin>>n;
        int x;
        for (int i=0; i<n; i++)
        {
            cin>>x;
            s.insert(x);
        }
        if (s.size()<=1)
           {
              cout <<-1<<endl;
             // continue;
           }

        else
        {
            auto it=s.rbegin();
            it++;
            cout <<*it<<endl;
        }
    }
    return 0;
}
