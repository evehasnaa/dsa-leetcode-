int isvalid (string s)
{   int ctr=0;
    stack <char>st;
    for (char ch:s)
    {
        if (ch=='(')
                st.push(ch);
        else
        {

             if (!st.empty())
              {
                  st.pop();
                   ctr++;
              }


        }

    }
         return ctr;
}
