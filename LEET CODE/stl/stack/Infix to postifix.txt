#include <iostream>
#include<stack>
#include<string>
using namespace std;
int prec (char c)
{
    if (c=='*'||c=='/')
        return 2;
    else if (c=='+'|| c=='-')
        return 1;
    else
        return 0;


}
string intopr(string t)
{
    stack<char>s;
    string posfix="";
    for (int i=0;i<t.size();i++)
    {
        if (isdigit(t[i]))
        {
             posfix+=t[i];
        }
        else
        {
            while (!s.empty()&& prec(s.top())>=prec(t[i]))
            {
                posfix+=s.top();
                s.pop();
            }
            s.push(t[i]);
        }
    }
    while (!s.empty())
    {
        posfix+=s.top();
        s.pop();
    }
    return posfix;
}
int main()
{
    string str="1+3*5-8/2";
    cout <<intopr(str)<<endl;

//13+58-*2/
    return 0;
}
