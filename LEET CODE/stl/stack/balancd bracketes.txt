int isvalid (string s)
{
    int ctr=0;
    stack <char>st;
    char x;
    for (int i=0; i<s.size(); i++)
    {
        if (s[i]=='('||s[i]=='['||s[i]=='{')
        {
            st.push(s[i]);
            continue;
        }

        if ( st.empty())
            return false;
        switch(s[i])
        {
        case ')':
                x=st.top();
               st.pop();

               if (x=='['||x=='{')
                return false ;
               break;
        case ']':
            x=st.top();
            st.pop();

            if (x=='('||x=='{')
                return false ;
            break;
        case '}':
                x=st.top();
                st.pop();

               if (x=='('||x=='[')
                return false ;
               break;



        }


    }

    if (st.empty())
        return 1;
    else
        return 0;
}
int main()
{
    string s;

    cin>>s;
    if (isvalid(s))
        cout <<"YES"<<endl;
    else
        cout <<"NO"<<endl;
    return 0;
}
