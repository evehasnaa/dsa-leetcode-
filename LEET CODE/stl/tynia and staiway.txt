#include <iostream>
#include<vector>

using namespace std;

int main()
{
   vector<int>ans;
   int n;
   cin>>n;
   int c=-1;
   int x;
   for (int i=0;i<n;i++)
   {
       cin>>x;

        if (x==1&& c!=-1)
        {
                ans.push_back(c);
        }
         c=x;
   }
    ans.push_back(c);
    cout <<ans.size()<<endl;
    for (int i=0;i<ans.size();i++)
   {
       cout <<ans[i]<<" ";
   }
   cout <<endl;
    return 0;
}
