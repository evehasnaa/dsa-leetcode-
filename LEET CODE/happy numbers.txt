int squareDigit(int n)
{
    
    int sum=0;
    while (n>0)
    { 
        int digit=n%10;
        sum+=digit*digit;
        n/=10;
    }return sum;
}

class Solution {
public:
    bool isHappy(int n) {
       
        set<int>s;
        while(s.find(n)==s.end() )
        {
              s.insert(n);
              n=squareDigit(n);
       }
      return n==1;
    }
};
