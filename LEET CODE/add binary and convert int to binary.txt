string addbinary (string a,string b)
{
    int i=a.size()-1;
    int j=b.size()-1;
    int sum=0,car=0;
    string res;

    while (i>=0||j>=0)
    {
        sum=car;
        if (i>=0) sum+=a[i--]-'0';
        if (j>=0) sum+=a[j--]-'0';

        res +=to_string(sum%2);
        car=sum/2;

    }
    if (car!=0) res+='1';
    reverse(res.begin(),res.end());
    return res;


}
